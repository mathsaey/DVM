37 # DVM Context matcher.
38 #
39 # The matcher is responsible for gathering inputs
40 # to the same instruction based on their contexts.
41 # The matcher should forward these tokens to the scheduler once
42 # that all of the inputs for a given context are available.
43 #
44 # The matcher utilizes <i>(instruction, context)</i> pairs as **keys**
45 ##
46 class ContextMatcher(object):
47 
48  ##
49  # Initialize a context matcher.
50  #
51  # \param core
52  # The runtime core that this matcher
53  # is a part of.
54  ##
55  def __init__(self, core):
56  super(ContextMatcher, self).__init__()
57 
58  ## Dictionary that contains the tokens for all the keys
59  self.tokens = {}
60 
61  ## Reference to the DVM::Core
62  self.core = core
63 
64  ##
65  # See if a given key is already a part of the
66  # tokens we are matching.
67  #
68  # Create an array to match these tokens if it's not.
69  #
70  # \param key
71  # The key to check
72  ##
73  def checkKey(self, key):
74  if key not in self.tokens:
75  inst = self.core.memory.get(key[0])
76  inputs = inst.inputs
77  arr = [None] * inputs
78  self.tokens.update({key : arr})
79 
80  ##
81  # Update the token array for a key
82  # This method assumes that there is an array for the
83  # given key.
84  #
85  # \param key
86  # The key to find the array we want to update.
87  # \param port
88  # The destination port of the token we want to add.
89  # \param token
90  # The token we want to add.
91  ##
92  def updateKeyArr(self, key, port, token):
93  arr = self.tokens[key]
94  arr[port] = token
95 
96  ##
97  # See if all the input tokens are present
98  # for a given key.
99  #
100  # \param key
101  # The key to check
102  #
103  # \return
104  # True if all the input tokens are present for key.
105  ##
106  def isKeyReady(self, key):
107  return self.tokens[key].count(None) == 0
108 
109  ##
110  # Add the tokens for key to the scheduler.
111  ##
112  def executeKey(self, key):
113  arr = self.tokens[key]
114  del self.tokens[key]
115  self.core.scheduler.schedule(key[0], arr)
116 
117  ##
118  # Add a token to the matcher
119  #
120  # This method can be seen as the main method of the
121  # matcher as it utilizes all the other methods to define
122  # the behaviour that the matcher will use when a token is added.
123  #
124  # \param token
125  # the token to add.
126  ##
127  def add(self, token):
128  log.info("mtch", "Adding token:", token)
129 
130  tag = token.tag
131  inst = tag.inst
132  cont = tag.cont
133  port = tag.port
134  key = (inst, cont)
135 
136  self.checkKey(key)
137  self.updateKeyArr(key, port, token)
138 
139  if self.isKeyReady(key):
140  self.executeKey(key)
