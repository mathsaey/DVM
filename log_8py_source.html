28 # \package log
29 # \brief DVM Logging
30 #
31 # Shared logger. The logger is responsible for printing output to the user.
32 # The logger is also built in such a way to allow multiple processes to use it
33 # without messing up the output.
34 #
35 # Heavy use of multiple processes is the reason for creating our own logger instead
36 # of using the standard library provided by python.
37 ##
38 
39 import StringIO
40 import multiprocessing
41 
42 ##
43 # Defines the various logging levels.
44 # These should be accessed by index.
45 #
46 # Key | Level | Explanation
47 # ----|-------|-------
48 # 0 | DEBG | Debug log message, the lowest possible value
49 # 1 | INFO | Info log message
50 # 2 | WARN | Warning message
51 # 3 | ERR! | Signal an error, users should expect things to start going wrong after seeing one of these.
52 ##
53 __severities__ = [
54  "\033[32mDEBG\033[0m",
55  "\033[34mINFO\033[0m",
56  "\033[33mWARN\033[0m",
57  "\033[31mERR!\033[0m"
58 ]
59 
60 ##
61 # The log that controls access to the output.
62 ##
63 __logLock__ = multiprocessing.Lock()
64 
65 ##
66 # Update the lock, this should happen after
67 # creating a new process.
68 ##
69 def setLock(lock):
70  global __logLock__
71  __logLock__ = lock
72 
73 ##
74 # Get a reference to the loglock.
75 ##
76 def getLock():
77  return __logLock__

79 ##
80 # Log a message.
81 #
82 # \param level
83 # The idx of the loglevel to use, see __severities__
84 # \param channel
85 # The name of the channel, try to stick to a short name,
86 # the channel should represent the part of dvm that is sending
87 # the message, such as the runtime or the instruction memory.
88 # \param message
89 # The message to send.
90 ##
91 def _log(level, channel, *msgLst):
92  buffer = StringIO.StringIO()
93 
94  buffer.write("[" + __severities__[level] + "]")
95  buffer.write("['" + channel + "']")
96  buffer.write("\t")
97 
98  for el in msgLst:
99  buffer.write(str(el))
100  buffer.write(" ")
101 
102  with __logLock__:
103  print buffer.getvalue()
104 
105  buffer.close()
106 
107 ## Log a debug severity message
108 def debg(channel, *msgLst): _log(0, channel, *msgLst)
109 ## Log an information message
110 def info(channel, *msgLst): _log(1, channel, *msgLst)
111 ## Log a warning message
112 def warn(channel, *msgLst): _log(2, channel, *msgLst)
113 ## Log an error message
114 def err(channel, *msgLst): _log(3, channel, *msgLst)
