58  # Get an output port of the subgraph.
59  # This maps to the ports of the entry node.
60  #
61  # This may seem counterintuitive, but this is seen
62  # from the perspective of the inside of the subgraph.
63  # In which case the output port is used to retrieve data
64  # from the subgraph.
65  ##
66  def getOutputPort(self, idx):
67  return self.entry.getOutputPort(idx)
68 
69  ##
70  # Gets an input port of the subgraph.
71  # This maps to the input of the return (exit) node.
72  #
73  # Once again this is seen from the inside of the subgraph.
74  # If we want to add data to the subgraph, we try to return it
75  # to the outside world, which is done through the exit node.
76  ##
77  def getInputPort(self, idx):
78  return self.exit.getInputPort(idx)
