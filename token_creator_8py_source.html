36 ##
37 # DVM Token creator.
38 #
39 # The token creator is responsible for
40 # wrapping results of instructions in tokens.
41 ##
42 class TokenCreator(object):
43  def __init__(self, core):
44  super(TokenCreator, self).__init__()
45  self.core = core
46 
47  self.contextMap = {}
48  self.restoreMap = {}
49 
50  ##
51  # Create a simple token, with a known destination.
52  # The token will remain on the current core.
53  ##
54  def simpleToken(self, datum, toInst, toPort, context):
55  tag = token.Tag(self.core.identifier, toInst, toPort, context)
56  tok = token.Token(datum, tag)
57  self.core.add(tok)
58  ##
59  # Change the context of a token.
60  # This will bind the context of this token to a
61  # certain return instruction.
62  #
63  # The runtime may decide to send these tokens
64  # to a different runtime core.
65  #
66  # Multiple tokens that are sent from the same instruction with the same
67  # context will receive the same context.
68  #
69  # \param token
70  # The token to modify. The destination port of this token
71  # will remain unchanged.
72  # \param toInst
73  # The instruction that will be the new destination of this
74  # token.
75  # \param retInst
76  # The return instruction that will be bound to the new context.
77  #
78  # \see restoreContext
79  ##
80  def changeContext(self, token, inst, toInst, retInst):
81  cont = token.tag.cont
82  key = (inst, cont)
83  new = None
84 
85  if key not in self.contextMap:
86  new = self.core.contextCreator.get()
87  self.contextMap.update({key : new})
88  self.restoreMap.update({new : (cont, retInst)})
89  else:
90  new = self.contextMap[key]
91 
92  token.tag.cont = new
93  token.tag.inst = toInst
94  self.core.add(token)
95 
96  ##
97  # Restore the old context of a token.
98  #
99  # In order to do this, we simply look up the previous context
100  # and the return instruction that are bound to this context.
101  ##
102  def restoreContext(self, token):
103  cont = token.tag.cont
104  pair = self.restoreMap[cont]
105 
106  token.tag.cont = pair[0]
107  token.tag.inst = pair[1]
108  self.core.add(token)
109 
110  def stopToken(self, tok):
111  tok.tag = token.StopTag()
112  self.core.add(tok)
