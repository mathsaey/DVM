82 ##
83 # A static instruction will always send it's
84 # output to the same instructions.
85 #
86 # A static instruction can have multiple output ports.
87 # Furthermore, multiple destinations can exist for a single port.
88 ##
89 class StaticInstruction(AbstractInstruction):
90  def __init__(self):
91  super(StaticInstruction, self).__init__()
92  self.destinations = {}
93 
94  ##
95  # Add a destination to this instruction.
96  #
97  # \param port
98  # The output port to link *from*
99  # \param toInst
100  # The instruction to send to
101  # \param toPort
102  # The port on this instruction to send to.
103  ##
104  def addDestination(self, port, toInst, toPort):
105  if port in self.destinations:
106  self.destinations[port] += [(toInst, toPort)]
107  else:
108  self.destinations.update({port : [(toInst, toPort)]})
109 
110  ##
111  # Send a datum to any destination of a given output port.
112  #
113  # \param datum
114  # The piece of data to send.
115  # \param core
116  # The currently active core.
117  # \param port
118  # The port that we send outputs from.
119  # \param cont
120  # The context of the output.
121  ##
122  def sendDatum(self, datum, core, port, cont):
123  for dst in self.destinations[port]:
124  inst = dst[0]
125  port = dst[1]
126  core.tokenCreator.simpleToken(
127  datum, inst, port, cont)

145  ##
146  # Send results to the relevant destinations.
147  #
148  # Simply forwards any element in the list to all the
149  # destinations of the matching output port of the instruction.
150  # The length of results should be equal to the amount of output ports.
151  ##
152  def sendResults(self, results, core, cont):
153  for i in xrange(0, len(results)):
154  res = results[i]
155  self.sendDatum(res, core, i, cont)

157  def execute(self, tokens, core):
158  log.info("inst", self, "executing.")
159  lst = map(lambda x : x.datum, tokens)
160  res = self.operation(*lst)
161  cont = tokens[0].tag.cont
162  self.sendResults([res], core, cont)
