134 ##
135 # Wrapper around a combined type that contains multiple base types
136 #
137 # The wrapper represents the full type starting at it's index.
138 # So if we have a tuple (int, str, int), then we would have 3
139 # _CombinedType instances, one that represents the full type,
140 # one that represents (str, int) and one that represents (int).
141 ##
142 class _CombinedType(_Type):
143  def __init__(self, type, containerType, next = None):
144  super(_CombinedType, self).__init__()
145  self.type = containerType
146  self.list = [type]
147  if next:
148  self.list += next.list
149 
150  def __str__(self):
151  res = "Combined: (" + self.type + ")"
152  for el in self.list:
153  res += " <" + str(el) + ">"
154  return res

223 def parseType(arr, ctr):
224  funcKey = int(arr[_code_idx])
225  try:
226  func = _type_codes[funcKey]
227  except KeyError:
228  tools.warning("Unknown type code: " + str(funcKey) + " encountered.", ctr)
229  _pool.addType(arr, _UnknownType())
230  else:
231  func(arr)

234  base_type = _basic_types[int(arr[_arg_1_idx])]

237 def _parseFunction(arr):
238  args = getType(arr[_arg_1_idx])
239  res = getType(arr[_arg_2_idx])
240  _pool.addType(arr, _FunctionType(args, res))

246 def _parseCombinedPtr(arr, container):

250 def _parseCombined(arr, container):
251  baseType = getType(arr[_arg_1_idx])
252 
253  # Add the new type, if there is a previous, link to it
254  try:
255  next = getType(arr[_arg_2_idx])
256  except KeyError:
257  _pool.addType(arr, _CombinedType(baseType, container))
258  else:
259  _pool.addType(arr, _CombinedType(baseType, container, next))
