36 # Represents a DVM token.
37 # A token carries program data and a Tag.
38 # The tag contains all the "meta" information about
39 # the token, such as it's destination and context.
40 ##
41 class Token(object):
42  def __init__(self, datum, tag):
43  super(Token, self).__init__()
44  self.datum = datum
45  self.tag = tag

47  def __str__(self):
48  tagString = "[" + str(self.tag) + "]"
49  dataString = "'" + str(self.datum) + "'"
50  return "<| " + dataString + " " + tagString + " |>"

61 ##
62 # Standard tag.
63 # A standard tag contains the destination
64 # (instruction and port) of a token as well as it's context.
65 #
66 # Conceptually, a tag has 2 parts:
67 # * A static part, it's port and instruction which are part of the program
68 # * A dynamic part, it's core and context, which are dynamically assigned at runtime.
69 ##
70 class Tag(object):
71  def __init__(self, core, inst, port, cont):
72  super(Tag, self).__init__()
73  self.core = core
74  self.cont = cont
75  self.port = port
76  self.inst = inst
77 
78  def __str__(self):
79  core = "core " + str(self.core) + " | "
80  inst = "inst " + str(self.inst) + " | "
81  port = "port " + str(self.port) + " | "
82  cont = "cont " + str(self.cont)
83  return core + inst + port + cont
84 
85  def isSpecial(self): return False
