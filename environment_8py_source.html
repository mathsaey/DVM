43 # Single part of the scope stack.
44 #
45 # A scope contains all of the definitions
46 # of the current scope.
47 #
48 ##
49 class Scope(object):
50 
51  ##
52  # Create a scope belonging to a subgraph.s
53  # By convention, the subgraph can always be
54  # found at label 0.
55  ##
56  def __init__(self, subGraph = None):
57  super(Scope, self).__init__()
58  self.nodes = [subGraph]
59 
60  ##
61  # Add a node to the scope.
62  #
63  # The label of the node should match
64  # the current length of the node list of this object.
65  # This is the case since IF1 nodes start are sequentially
66  # numbered starting from one.
67  #
68  # \param label
69  # The IF1 label of the node.
70  # \param node
71  # The IGR node to assign to the label
72  ##
73  def addNode(self, label, node):
74  if len(self.nodes) is label:
75  self.nodes += [node]
76  else:
77  # This situation can occur in IF1 (might be a bug)
78  # so fill the list with none until the length is ok.
79  tools.warning("Non-sequential node label added!")
80  padding = [None for i in xrange(0, label - len(self.nodes))]
81  self.nodes += padding
82  self.addNode(label, node)
83

85  # Get a node from the scope.
86  #
87  # \param label
88  # The label of the node to get
89  ##
90  def getNode(self, label):
91  return self.nodes[label]
92 
93  ##
94  # Get the subgraph of the scope.
95  ##
96  def getSubGraph(self):
97  return self.nodes[0]

121 ##
122 # A stack with the global scope at the bottom,
123 # and the current scope at the top.
124 ##
125 __STACK__ = [Scope()]

127 ##
128 # Create a new scope and push
129 # it on top of the stack.
130 ##
131 def scope(subGraph):
132  global __STACK__
133  __STACK__ = [Scope(subGraph)] + __STACK__
