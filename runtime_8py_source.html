28 # \package dvm.runtime
29 # \brief DVM runtime core
30 #
31 # This module defines the runtime.
32 # The runtime is responsible for dispatching tokens,
33 # matching tokens by their contexts and for scheduling
34 # instructions that are ready to execute.
35 #
36 # Multiple runtime "cores" are active at any given time, depending
37 # on the system. It is the responsibility of the runtime to find a
38 # decent load balance accross these cores.
39 ##

55 ##
56 # Runtime core.
57 #
58 # A runtime core is a worker unit in DVM.
59 # It defines it's own scheduler, matcher and
60 # token dispatcher and it has a static copy of the
61 # instruction memory.
62 ##
63 class Core(object):
64  ##
65  # Initialize a core.
66  #
67  # This method only initializes all the components of the
68  # core that are not dependent on multiprocessing elements.
69  #
70  # \param identifier
71  # The identifier of this core, this identifier should be unique
72  # and it should match the identifier of this core in the collection
73  # of all the cores.
74  # \param instructions
75  # A reference to the static instruction memory.
76  ##
77  def __init__(self, identifier, memory):
78  super(Core, self).__init__()
79  log.info("core", "Initializing core:", identifier)
80 
81  ## Instruction memory
82  self.memory = memory
83  ## Identifier of this core. (integer)
84  self.identifier = identifier
85  ## See if this core is running.
86  self.active = True
87  ## Message Queue of this core
88  self.inbox = None
89  ## Message Queues of the other cores
90  self.cores = None
91 
92  ## Context creator for this core
93  self.contextCreator = ContextCreator(self)
94  ## Token creator for this core
95  self.tokenCreator = TokenCreator(self)
96  ## Token dispatcher for this core
97  self.dispatcher = TokenDispatcher(self)
98  ## Scheduler for this core
99  self.scheduler = Scheduler(self)
100  ## Context matcher for this core
101  self.matcher = ContextMatcher(self)
102 
103  ## String representation of a core.
104  def __str__(self):
105  return "Core: " + str(self.identifier)
106 
107  ## See if 2 cores are equal
108  def __eq__(self, other):
109  return self.identifier == other.identifier
110 
111  ##
112  # Add a token to the inbox of a core.
113  #
114  # \param token
115  # The token to add.
116  # \param core
117  # The core to add the token to.
118  # The current core will be used if
119  # this argument is not added.
120  #
121  def add(self, token, core = None):
122  if core:
123  self.cores[core].put(token)
124  else:
125  self.inbox.put(token)
126 
127  ##
128  # Set up the multiprocessing elements
129  # of the core and start the runtime.
130  #
131  # \param inbox
132  # The message queue of this corec.
133  # \param cores
134  # A list of the message queues of the
135  # other cores
136  # \param logLock
137  # The lock of the logger.
138  ##
139  def start(self, inbox, cores, logLock):
140  log.setLock(logLock)
141  self.inbox = inbox
142  self.cores = cores
143 
144  log.info("core", self, "Starting run loop")
145 
146  while self.active:
147  t = self.inbox.get()
148  self.dispatcher.process(t)
149 
150  ##
151  # Stop the current core.
152  #
153  # \param value
154  # The result that the program
155  # returned.
156  ##
157  def stop(self, value):
158  log.info("core", self, "Terminated with value:", value)
159  self.active = False
160  print value
161  return value
162 
163 ##
164 # Initialize the cores, and start program execution.
165 #
166 # \param cores
167 # The amount of cores to create.
168 ##
169 def start(cores = 1, tokens = []):
170  logLock = log.getLock()
171  coreLst = [Core(i, memory.memory()) for i in xrange(0, cores)]
172  queues = [multiprocessing.Queue() for i in xrange(0, cores)]
173 
174  for i in xrange(0, cores):
175  core = coreLst[i]
176  queue = queues[i]
177 
178  p = multiprocessing.Process(
179  target = core.start,
180  args = (queue, queues, logLock))
181  p.start()
182 
183  #-- DEBUG --#
184  for t in tokens:
185  queues[0].put(t)
