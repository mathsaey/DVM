66  # Create a new node.
67  #
68  # \param subGraph
69  # The subgraph this node belongs to.
70  # \param inputs
71  # The amount of inputs this node will accept
72  # This argument is optional, the input amount
73  # will be updated depending on the amount of
74  # incoming edges.
75  # \param outputs
76  # The amount of outputs this node will return.
77  # The same rules that apply to inputs apply here.
78  ##
79  def __init__(self, subGraph, inputs = 0, outputs = 0):
80  super(Node, self).__init__()
81  self.subGraph = subGraph
82  self.key = getKey()
83  self.inputs = inputs
84  self.outputs = outputs
85  self.inputPorts = [port.InputPort(self, i) for i in xrange(0,inputs)]
86  self.outputPorts = [port.OutputPort(self, i) for i in xrange(0,outputs)]
87 
88  ##
89  # Create a printable version of the node
90  ##

92  name = self.__class__.__name__
93  return name + " " + "'" + str(self.key) + "'"

95  # Gets an input port. Create it if it doesn't exist yet.
96  # This allows us to determine the amount of inputs
97  # a certain node will have, even if this is not explicit in IF1
98  #
99  # \param idx
100  # The idx of the port you need
101  # \return
102  # The port at idx
103  ##
104  def getInputPort(self, idx):
105  try:
106  res = self.inputPorts[idx]
107  return res
108  except IndexError:
109  self.inputs += 1
110  self.inputPorts += [port.InputPort(self, idx)]
111  return self.getInputPort(idx)
