93 ##
94 # Create a new subgraph
95 #
96 # \param inputs
97 # The amount of inputs the subgraph accepts
98 # \param outputs
99 # The amount of data the subgraph returns
100 # \return
101 # The subgraph. It's entry and exit fields
102 # should be used to access parameters and
103 # return values of this subgraph.
104 #
105 def createGeneralSubGraph(name , inputs, outputs):
106  subGraph = subgraph.SubGraph(None, None, name)
107  entry = node.SubGraphEntryNode(subGraph, inputs)
108  exit = node.SubGraphExitNode(subGraph, outputs)
109  subGraph.entry = entry
110  subGraph.exit = exit
111  subGraph.addNode(entry)
112  subGraph.addNode(exit)
113  return subGraph
114 
115 ##
116 # Create a subgraph and add it to the program graph.
117 ##
118 def createSubGraph(name, inputs, outputs):
119  subGraph = createGeneralSubGraph(name, inputs, outputs)
120  graph.addSubGraph(subGraph)
121  graph.bindName(subGraph)
122  return subGraph
123

125 # Create a subgraph for a compound node.
126 ##
