74 ##
75 # Get a representation for a port.
76 #
77 # \return
78 # Return the value of the literal, if this port
79 # accepts a literal. * if this port is connected to
80 # another port. Returns the empty string if this port
81 # is not connected to anything
82 ##
83 def portString(port, preFix = ""):
84  if port.acceptsLiteral(): return str(port.source.value)
85  elif port.isConnected(): return "<" + preFix + str(port.idx) + ">" + " *"
86  else: return ""
87 
88 ##
89 # String representation of a port list.
90 #
91 # \param portLst
92 # The list with ports, should not be None
93 #
94 # \return
95 # Returns a string that will show the values of all the ports
96 # in a horizontal line when parsed by dot.
97 # Returns the empty string if the portLst is None
98 ##
99 def ports(portLst, prefix = ""):
100  res = ""
101  for port in portLst:
102  res += "|" + portString(port, prefix)
103  return "{" + res[1:] + "}"

105 ## Get the portlist for the inputs of a node.
106 def inputList(node):
107  if node.inputPorts:
108  return ports(node.inputPorts, "I") + "|"

111 ## Get the portlist for the outputs of a node.
112 def outputList(node):

125 ## Convert a connection to a string
126 def edgeStr(srcNode, scrPort, dstNode, dstPort):
127  src = nodeIdentifier(srcNode) + " : O" + str(scrPort)
128  dst = nodeIdentifier(dstNode) + " : I" + str(dstPort)
129  return src + " -> " + dst + ";"

131 ## Add the label of the node to the buffer.
132 def nodeLabel(buffer, node):
133  buffer.write(nodeIdentifier(node))
134  buffer.write(' [')
135  buffer.write('label="' + '{' + inputList(node) + str(node) + outputList(node) + '}"')
136  if node.isCompound(): buffer.write("style = filled, fillcolor = lightgrey")

140 ## Add all the outgoing edges of a node to the buffer.

142  if node.hasNext():
143  for port in node.outputPorts:
144  for target in port.targets:

163 def dotFooter(buffer):

166 ## Create the dot string
167 def getDot(skipCompound):
168  buffer = StringIO.StringIO()
169  dotHeader(buffer)
170 
171  traverse.traverseAll(
172  lambda nd: node(buffer, nd),
173  lambda sg: subGraphHeader(buffer, sg),
174  lambda sg: subGraphFooter(buffer, sg),
175  skipCompound,
176  lambda nd: compoundHeader(buffer, nd),
177  lambda nd: compoundFooter(buffer, nd)
178  )
179 
180  dotFooter(buffer)
181  str = buffer.getvalue()
182  buffer.close()
183  return str

193 ##
194 # Convert the IGR graph to dot, save it,
195 # and run dot on this file.
196 #
197 # This function should be call with keyword arguments.
198 # The default arguments will cause the following behaviour:
199 # * dot is assumed to be in your PATH.
200 # * the dot file will be saved in igr.dot
201 # * the output will be in png format.
202 # * dot will decide where to store the output.
203 # With the default settings this would be in igr.dot.png
204 #
205 # \param dotpath
206 # The path of the dot executable, in case it's not in your PATH
207 # \param path
208 # The location where the dot file will be stored.
209 # \param format
210 # The output format of the graph dot creates from the dot file.
211 # \param output
212 # The location where we store the output of dot.
213 # Leaving this blank will pass the -O option.
214 # The -O option let's dot choose the path.
215 # \param other
216 # Any other options you want to pass to doth.
217 # These options should be passed as a list of strings.
218 # \param skipCompound
219 # True if you do not want to display the compound nodes.
220 ##
221 def runDot(
222  dotpath = "dot",
223  path = "igr.dot",
224  format = "png",
225  output = "",
226  other = [],
227  skipCompound = True
228  ):
229 
230  dotToFile(path, skipCompound)
231 
232  format = "-T" + format
233 
234  if output: output = "-o" + output
235  else: output = "-O"
236 
237  subprocess.check_call([dotpath, format, path, output] + other)
