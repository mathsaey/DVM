128 ##
129 # Parse the start of a compound node.
130 #
131 # We do this by creating a compound scope to
132 # store the subgraphs. We also tell the environment
133 # that we entered a compound node.
134 #
135 # Finally, we add a dummy node that will be popped
136 # by the first subgraph we meet.
137 ##
138 def parseCompoundStart(arr, ctr):
139  environment.scopeCompound()
140  environment.enterComp()
141  environment.scope(None)

143 def parseCompoundEnd(arr, ctr):
144  # Enter the compound scope
145  environment.popScope()
146 
147  # get and order the subgraphs of the compound node.
148  length = int(arr[_ce_len_idx])
149  endIdx = _ce_lis_idx + length
150  assocLst = arr[_ce_lis_idx:endIdx]
151  subGraphs = environment.getSubGraphs()
152  resGraphs = []
153 
154  for idx in assocLst:
155  graph = subGraphs[int(idx)]
156  graph.name = str(idx)
157  resGraphs += [graph]
158 
159  # Restore the environment
160  environment.exitComp()
161  environment.popScope()
162 
163  # Create the compound node
164  compType = arr[_ce_code_idx]
165  const = compound.getCompound(compType, ctr)
166  node = igr.createCompoundNode(
167  const,
168  environment.getSubGraph(),
169  resGraphs)
170 
171  # Add the node to the environment.
172  label = int(arr[_ce_label_idx])
173  environment.addNode(label, node)
